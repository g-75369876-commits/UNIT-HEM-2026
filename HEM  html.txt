<script>
    // Initialize Icons
    lucide.createIcons();

    // Set Current Date
    const d = new Date();
    document.getElementById('current-date').innerText = d.toLocaleDateString('ms-MY', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

    // Page Switching Logic
    function showPage(pageId) {
        // Hide all pages
        const pages = document.querySelectorAll('.page');
        pages.forEach(p => p.classList.remove('active'));

        // Remove active class from nav
        const navs = document.querySelectorAll('.nav-item');
        navs.forEach(n => n.classList.remove('active'));

        // Show selected page
        document.getElementById(pageId).classList.add('active');

        // Update Title in Header
        const titles = {
            'main-dashboard': 'Dashboard Utama HEM',
            'page-induk': 'Jawatankuasa Induk HEM',
            'page-disiplin': 'Unit Disiplin & SSDM',
            'page-spbt': 'Unit SPBT',
            'page-rmt': 'Unit RMT',
            'page-apdm': 'Data Maklumat Murid (APDM)',
            'page-3k': 'Unit 3K (Keselamatan, Kesihatan, Keceriaan)'
        };
        document.getElementById('current-title').innerText = titles[pageId];
    }
</script>
